///////////////////////////////////////////////
Classement des Eleves
///////////////////////////////////////////////

SELECT E.ID_Joueur, E.nom_Joueur, 3*E.Victoires_Joueur + E.Nuls_Joueur AS "Points"
FROM Eleve E
ORDER BY Points DESC

/*********************************************************************************************************************************/
///////////////////////////////////////////////
TEMPS DE PARTIE
///////////////////////////////////////////////

SELECT E.ID_Joueur, E.nom_Joueur, SUM(P.Temps_Partie) AS "Temps_Total"
FROM Eleve E, Partie P

WHERE
E.ID_Joueur = P.ID_JoueurBlancPartie OR E.ID_Joueur = P.ID_JoueurNoirPartie

GROUP BY E.ID_Joueur
ORDER BY Temps_Total
